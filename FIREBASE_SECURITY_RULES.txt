// Firestore Security Rules for YouTube Search Cache
// Copy and paste these rules into Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Allow read-only access to API keys (for app usage)
    match /config/apiKeys {
      allow read: if true;
      allow write: if false; // Only you can write via console
    }
    
    // Allow read/write access to searchCache collection
    // Anyone can read and write, making it a public cache
    match /searchCache/{searchQuery} {
      allow read: if true;  // Anyone can read cached searches
      allow write: if true; // Anyone can write to cache (public caching)
    }
    
    // Online users tracking (for real-time user count) - 개선된 보안 규칙
    match /onlineUsers/{uid} {
      allow read: if true;  // Anyone can read online status (for user count display)
      allow write: if request.auth != null && request.auth.uid == uid;  // Only owner can write
    }
    
    // User-specific data (e.g., favorites, recent searches, filter presets, username)
    match /users/{uid} {
      allow read: if request.auth != null && request.auth.uid == uid;
      allow write: if request.auth != null && request.auth.uid == uid;
    }
    
    // Optional: Restrict to authenticated users only
    // Uncomment the following rules if you want authentication:
    /*
    match /searchCache/{searchQuery} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    */
    
    // Deny all other access by default
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
